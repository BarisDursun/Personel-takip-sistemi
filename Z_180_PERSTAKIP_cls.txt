CLASS lcl_class_0200 DEFINITION.
  PUBLIC SECTION.
    METHODS: start_screen,
             pbo_0200,
             pai_0200 IMPORTING iv_ucomm TYPE sy-ucomm,
             get_data,
             set_fcat,

             display_alv.
ENDCLASS.

CLASS lcl_class_0100 IMPLEMENTATION.

  METHOD start_screen.
    CALL SCREEN 0100.
  ENDMETHOD.

  METHOD pbo_0100.
    SET PF-STATUS '0100'.
    go_main_0100->set_fcat( ).
    go_main_0100->set_layout( ).
    go_main_0100->display_alv( )..
  ENDMETHOD.

  METHOD pai_0100.
    CASE iv_ucomm.
      WHEN '&BACK'.
        SET SCREEN 0.
      WHEN '&SAVE'.
        go_main_0100->check_data( ).
      WHEN OTHERS.
    ENDCASE.
  ENDMETHOD.

  METHOD check_data.
    LOOP AT gt_alvtable_0100 ASSIGNING FIELD-SYMBOL(<fs_alv>).
      " daha önce işlendimi
      IF <fs_alv>-personel_id IS NOT INITIAL
        AND <fs_alv>-personel_ad IS NOT INITIAL
        AND <fs_alv>-personel_soyad IS NOT INITIAL
        AND <fs_alv>-personel_departman_id IS NOT INITIAL
        AND <fs_alv>-personel_title IS NOT INITIAL
        AND <fs_alv>-toplam_tecrube IS NOT INITIAL
        AND <fs_alv>-line_color NE 'C510'. " C510 değilse (işlenmemişse) devam et

        " Unvan tablosundan kontrol yapılıyor
        SELECT SINGLE * FROM z180_titleblg_t
          INTO @DATA(ls_ttlblg)
          WHERE title_id EQ @<fs_alv>-personel_title.

        " Departman tablosundan kontrol yapılıyor
        SELECT SINGLE * FROM z180_dept_t
          INTO @DATA(ls_depblg)
          WHERE departman_id EQ @<fs_alv>-personel_departman_id.

        " eğer verilerin karşılığı varsa insert at
        IF ls_ttlblg IS NOT INITIAL AND ls_depblg IS NOT INITIAL.
          go_main_0100->insert_data( iv_alvtable = <fs_alv> ).
          <fs_alv>-line_color = 'C510'. "başarılı ise yeşil yap.
        ENDIF.
      ENDIF.
    ENDLOOP.
    go_main_0100->disable_editing( ). "işlemden sonra düzeltme kapat.
  ENDMETHOD.

  METHOD set_fcat.
    CLEAR gt_fcat.

    gs_fcat-ref_table = 'Z180_PERSBLG_T'.
    gs_fcat-ref_field = 'PERSONEL_ID'.
    gs_fcat-fieldname = 'PERSONEL_ID'.
    APPEND gs_fcat TO gt_fcat.
    CLEAR: gs_fcat.

    gs_fcat-ref_table = 'Z180_PERSBLG_T'.
    gs_fcat-ref_field = 'PERSONEL_AD'.
    gs_fcat-fieldname = 'PERSONEL_AD'.
    APPEND gs_fcat TO gt_fcat.
    CLEAR: gs_fcat.

    gs_fcat-ref_table = 'Z180_PERSBLG_T'.
    gs_fcat-ref_field = 'PERSONEL_SOYAD'.
    gs_fcat-fieldname = 'PERSONEL_SOYAD'.
    APPEND gs_fcat TO gt_fcat.
    CLEAR: gs_fcat.

    gs_fcat-ref_table = 'Z180_PERSBLG_T'.
    gs_fcat-ref_field = 'PERSONEL_DEPARTMAN_ID'.
    gs_fcat-fieldname = 'PERSONEL_DEPARTMAN_ID'.
    APPEND gs_fcat TO gt_fcat.
    CLEAR: gs_fcat.

    gs_fcat-ref_table = 'Z180_PERSBLG_T'.
    gs_fcat-ref_field = 'PERSONEL_TITLE'.
    gs_fcat-fieldname = 'PERSONEL_TITLE'.
    APPEND gs_fcat TO gt_fcat.
    CLEAR: gs_fcat.

    gs_fcat-ref_table = 'Z180_PERSBLG_T'.
    gs_fcat-ref_field = 'TOPLAM_TECRUBE'.
    gs_fcat-fieldname = 'TOPLAM_TECRUBE'.
    APPEND gs_fcat TO gt_fcat.
    CLEAR: gs_fcat.
  ENDMETHOD.

  METHOD set_layout.
    CLEAR gs_layout.
    gs_layout-zebra = abap_true.
    gs_layout-cwidth_opt = abap_true.
    gs_layout-col_opt = abap_true.
    gs_layout-edit = abap_true.
    gs_layout-info_fname = 'LINE_COLOR'.
    gs_layout-stylefname = 'IS_EDITABLE'. "dinamik olarak disable yapma
  ENDMETHOD.

  METHOD display_alv.
    IF go_grid IS INITIAL.
      CREATE OBJECT go_container
        EXPORTING container_name = 'CC_ALV'.

      CREATE OBJECT go_grid
        EXPORTING i_parent = go_container.

      go_grid->set_table_for_first_display(
        EXPORTING is_layout       = gs_layout
        CHANGING  it_outtab       = gt_alvtable_0100
                  it_fieldcatalog = gt_fcat ).

      go_grid->register_edit_event( "user edit yaptığında hemen tetikler modify
        EXPORTING i_event_id = cl_gui_alv_grid=>mc_evt_modified ).
    ELSE.
      CALL METHOD go_grid->refresh_table_display. " birdahikinde refreh eder
    ENDIF.
  ENDMETHOD.

  METHOD insert_data.
    DATA: ls_insertdata TYPE z180_persblg_t.
    ls_insertdata-personel_id            = iv_alvtable-personel_id.
    ls_insertdata-personel_ad            = iv_alvtable-personel_ad.
    ls_insertdata-personel_soyad         = iv_alvtable-personel_soyad.
    ls_insertdata-personel_departman_id  = iv_alvtable-personel_departman_id.
    ls_insertdata-personel_title         = iv_alvtable-personel_title.
    ls_insertdata-toplam_tecrube         = iv_alvtable-toplam_tecrube.
    INSERT z180_persblg_t FROM ls_insertdata.
    COMMIT WORK.
  ENDMETHOD.

  METHOD disable_editing.
    DATA: lwa_styleline TYPE lvc_s_styl.
    LOOP AT gt_alvtable_0100 ASSIGNING FIELD-SYMBOL(<fs_dis>).
      IF <fs_dis>-line_color EQ 'C510'. "yeşil olnaları kiliitle
        lwa_styleline-fieldname = 'PERSONEL_ID'.
        lwa_styleline-style = cl_gui_alv_grid=>mc_style_disabled.
        INSERT lwa_styleline INTO TABLE <fs_dis>-is_editable.

        lwa_styleline-fieldname = 'PERSONEL_AD'.
        lwa_styleline-style = cl_gui_alv_grid=>mc_style_disabled.
        INSERT lwa_styleline INTO TABLE <fs_dis>-is_editable.

        lwa_styleline-fieldname = 'PERSONEL_SOYAD'.
        lwa_styleline-style = cl_gui_alv_grid=>mc_style_disabled.
        INSERT lwa_styleline INTO TABLE <fs_dis>-is_editable.

        lwa_styleline-fieldname = 'PERSONEL_DEPARTMAN_ID'.
        lwa_styleline-style = cl_gui_alv_grid=>mc_style_disabled.
        INSERT lwa_styleline INTO TABLE <fs_dis>-is_editable.

        lwa_styleline-fieldname = 'PERSONEL_TITLE_ID'.
        lwa_styleline-style = cl_gui_alv_grid=>mc_style_disabled.
        INSERT lwa_styleline INTO TABLE <fs_dis>-is_editable.

        lwa_styleline-fieldname = 'TOPLAM_TECRUBE'.
        lwa_styleline-style = cl_gui_alv_grid=>mc_style_disabled.
        INSERT lwa_styleline INTO TABLE <fs_dis>-is_editable.
      ENDIF.
    ENDLOOP.
  ENDMETHOD.

ENDCLASS.

CLASS lcl_class_0200 IMPLEMENTATION.

  METHOD start_screen.
    CLEAR: go_grid,
           go_container.
    CALL SCREEN 0200.
  ENDMETHOD.

  METHOD pbo_0200.
    SET PF-STATUS '0200'.
    go_main_0200->get_data( ).
    go_main_0200->set_fcat( ).
    go_main_0200->set_layout( ).
    go_main_0200->display_alv( ).
  ENDMETHOD.

  METHOD pai_0200.
    CASE iv_ucomm.
      WHEN '&BACK'.
        SET SCREEN 0.
    ENDCASE.
  ENDMETHOD.

  METHOD get_data.
    " Ada göre
    IF gso_pa IS NOT INITIAL.
      SELECT per~personel_id, per~personel_ad, per~personel_soyad,
             dep~departman_ad, ttl~title_ad, per~toplam_tecrube, ttl~maas_katsayisi
        FROM Z180_PERSBLG_T AS per
        INNER JOIN Z180_DEPT_T AS dep ON dep~departman_id EQ per~personel_departman_id "veri eşlemeleri yap
        INNER JOIN Z180_TITLEBLG_T AS ttl ON ttl~title_id EQ per~PERSONEL_TITLE
        WHERE per~personel_ad IN @gso_pa "adı bu olanları al
        INTO TABLE @DATA(lt_alvtable).
    ENDIF.

    " Soyad
    IF gso_psa IS NOT INITIAL.
      SELECT per~personel_id, per~personel_ad, per~personel_soyad,
             dep~departman_ad, ttl~title_ad, per~toplam_tecrube, ttl~maas_katsayisi
        FROM Z180_PERSBLG_T AS per
        INNER JOIN Z180_DEPT_T AS dep ON dep~departman_id EQ per~personel_departman_id
        INNER JOIN Z180_TITLEBLG_T AS ttl ON ttl~title_id EQ per~PERSONEL_TITLE
        WHERE per~personel_soyad IN @gso_psa
        INTO TABLE @DATA(lt_psa_alvtable).
    ENDIF.

    "Departman Adı
    IF gso_pda IS NOT INITIAL.
      SELECT per~personel_id, per~personel_ad, per~personel_soyad,
             dep~departman_ad, ttl~title_ad, per~toplam_tecrube, ttl~maas_katsayisi
        FROM Z180_PERSBLG_T AS per
        INNER JOIN Z180_DEPT_T AS dep ON dep~departman_id EQ per~personel_departman_id
        INNER JOIN Z180_TITLEBLG_T AS ttl ON ttl~title_id EQ per~PERSONEL_TITLE
        WHERE dep~departman_ad IN @gso_pda
        INTO TABLE @DATA(lt_pda_alvtable).
    ENDIF.

    " title adı
    IF gso_pta IS NOT INITIAL.
      SELECT per~personel_id, per~personel_ad, per~personel_soyad,
             dep~departman_ad, ttl~title_ad, per~toplam_tecrube, ttl~maas_katsayisi
        FROM Z180_PERSBLG_T AS per
        INNER JOIN Z180_DEPT_T AS dep ON dep~departman_id EQ per~personel_departman_id
        INNER JOIN Z180_TITLEBLG_T AS ttl ON ttl~title_id EQ per~PERSONEL_TITLE
        WHERE ttl~title_ad IN @gso_pta
        INTO TABLE @DATA(lt_pta_alvtable).
    ENDIF.

    "ana tabloya ekle dataları:
    LOOP AT lt_psa_alvtable INTO DATA(ls_psa_alvtable).
      APPEND ls_psa_alvtable TO lt_alvtable.
    ENDLOOP.
    LOOP AT lt_pda_alvtable INTO DATA(ls_pda_alvtable).
      APPEND ls_pda_alvtable TO lt_alvtable.
    ENDLOOP.
    LOOP AT lt_pta_alvtable INTO DATA(ls_pta_alvtable).
      APPEND ls_pta_alvtable TO lt_alvtable.
    ENDLOOP.

    SORT lt_alvtable ASCENDING BY personel_id. "dimplicates çalışması için aynı idler alt alta olmalı.
    DELETE ADJACENT DUPLICATES FROM lt_alvtable COMPARING personel_id. "aynı id duplicates siler

    "temiz datada loop
    loop at lt_alvtable into data(ls_alvtable). "ham verileri table a koy
      CLEAR gs_alvtable_0200.
      gs_alvtable_0200-personel_id            = ls_alvtable-personel_id.
      gs_alvtable_0200-personel_ad            = ls_alvtable-personel_ad.
      gs_alvtable_0200-personel_soyad         = ls_alvtable-personel_soyad.
      gs_alvtable_0200-personel_departman_ad  = ls_alvtable-departman_ad.
      gs_alvtable_0200-personel_title_ad      = ls_alvtable-title_ad.
      gs_alvtable_0200-toplam_tecrube         = ls_alvtable-toplam_tecrube.
      gs_alvtable_0200-personel_maas          = ls_alvtable-toplam_tecrube * ls_alvtable-maas_katsayisi * 2000. "Maas hesap

      " Yıllık İzin için
      IF ls_alvtable-toplam_tecrube LT 1.
        gs_alvtable_0200-personel_yillik_izin = 'YOK'.
      ELSEIF ls_alvtable-toplam_tecrube LT 3.
        gs_alvtable_0200-personel_yillik_izin = '1 Hafta'.
      ELSEIF ls_alvtable-toplam_tecrube LT 5.
        gs_alvtable_0200-personel_yillik_izin = '2 Hafta'.
      ELSEIF ls_alvtable-toplam_tecrube LT 10.
        gs_alvtable_0200-personel_yillik_izin = '3 Hafta'.
      ELSE.
        gs_alvtable_0200-personel_yillik_izin = '1 Ay'.
      ENDIF.

      INSERT gs_alvtable_0200 INTO TABLE gt_alvtable_0200.
    ENDLOOP.
  ENDMETHOD.

    CLEAR gs_fcat.
